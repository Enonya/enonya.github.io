    #include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

#define meow(x) cerr << #x << " = " << x

using namespace std;
using i64 = long long;

struct KJGKMTZB {
    char tql;
    int orz;
};

int main() {    

    cin.tie(0) -> sync_with_stdio(false);
    cin.exceptions(cin.failbit | cin.badbit);

// --------------------------------------------------------------

    int a[] = {1, 2, 3, 4, 5};
    int b = 12345;

    int* p = a; 
    // 定义指针的时候要和指向的东西的类型相匹配。
    // 指针的类型是对应类型名后加一个 * 号
    // 数组名就是这个数组的第一个元素的地址，不用&

    int* q = &b;

// --------------------------------------------------------------


    p[2] = 7;   
    for(auto x : a) meow(x) << " "; 
    cerr << endl;

    // 赋值也可以直接这么写：

    p = a, p = p + 2, *p = 8;   
    for(auto x : a) meow(x) << " "; 
    cerr << endl;
    // 因为 p 指向这个数组的第一个元素，可以利用指针偏移来访问数组元素。
    // 数组的 [] 运算符就是这个原理。

    *q = 114514;
    meow(sizeof(q)) << endl; // 指针毕竟指向的是一块空间，所以也是可以 sizeof 的。
    meow(b) << endl; // 直接用指针解引用之后修改了这块空间对应的值

// --------------------------------------------------------------


    int* kot = new int[2]; // 动态申请一块空间
                           // 此处是申请一个长度为 2 的 int 数组，所以最后删除的时候
                           // 需要使用 delete[] 运算符。
    kot[0] = 4, kot[1] = 2;
    for(int i = 0; i <= 1; ++i)
        meow(kot[i]) << " "; 
    cerr << endl;
    // kot[] 相当于利用 [] 运算符实现指针偏移。
    // 它会自动解引用

    delete[] kot;


// --------------------------------------------------------------


    KJGKMTZB* yhp = new KJGKMTZB;
    yhp -> tql = 'a';   // 用指针访问成员
    yhp -> orz = 114514;

    cout << yhp -> tql << " " << yhp -> orz << endl;

    KJGKMTZB rmp = (KJGKMTZB){'k', 19260817};
    (*yhp) = rmp; // 可以直接解引用之后赋值。

    cout << yhp -> tql << " " << yhp -> orz << endl;

// --------------------------------------------------------------

    delete yhp;
    // delete 完之后 yhp 是个野指针。

    yhp = nullptr;
    // delete 完之后，如果之后可能用到，就先赋值一个 nullptr.
    delete yhp; // 对 nullptr delete 是合法的。

    if(yhp == nullptr) cout << "/ll" << endl;

// --------------------------------------------------------------

    int* this_is_a_bug; // 未初始化的野指针
    delete this_is_a_bug; // 这样写是要出事的。


    KJGKMTZB* pt;
    for(int i = 1; i <= 2; ++i) {
        KJGKMTZB qaq; // 这是一个临时变量
        pt = &qaq;
    }
    // for 循环退出之后 qaq 的生命周期就结束了。

    pt -> orz = 114514; // 此时 pt 指向的临时变量已经被释放了，所以 pt 也是一个野指针。
                        // 这样写是要出事的。
// --------------------------------------------------------------

// 函数指针 oi 里用的不多，暂时不写。

    return 0;
}
